    document.getElementById('chat-input').value = `Explain ${v} using the Day 1 curriculum.`;
    switchTab('assistant');
    sendMessage();
    };
    cloud.appendChild(span);
    });
    }
    // AI Logic
    async function sendMessage() {
    const input = document.getElementById('chat-input');
    const userText = input.value.trim();
    if (!userText) return;
    const chatBox = document.getElementById('chat-messages');
    const userDiv = document.createElement('div');
    userDiv.className = "flex justify-end animate-fade";
    userDiv.innerHTML = `<div class="chat-bubble-user max-w-[85%] p-4 text-sm">${userText}</div>`;
    chatBox.appendChild(userDiv);
    input.value = "";
    chatBox.scrollTop = chatBox.scrollHeight;
    const aiLoadingDiv = document.createElement('div');
    aiLoadingDiv.className = "flex justify-start animate-fade";
    aiLoadingDiv.innerHTML = `<div class="chat-bubble-ai max-w-[85%] p-4 text-sm italic text-stone-400">Consulting Day 1 transcripts...</div>`;
    chatBox.appendChild(aiLoadingDiv);
    chatBox.scrollTop = chatBox.scrollHeight;
    try {
    const response = await callGemini(userText);
    aiLoadingDiv.innerHTML = `<div class="chat-bubble-ai max-w-[85%] p-4 text-sm">${response}</div>`;
    } catch (error) {
    aiLoadingDiv.innerHTML = `<div class="chat-bubble-ai max-w-[85%] p-4 text-sm text-red-500">The local hub connection is weak. Try again.</div>`;
    }
    chatBox.scrollTop = chatBox.scrollHeight;
    }
    async function callGemini(query, retries = 0) {
    const systemPrompt = `You are "The Localization Assistant," an AI mentor focusing on Day 1 of the Localization Movement curriculum.
    Focus: The Global Shadow, The Metacrisis, and Ecological Overshoot.
    Core Metaphors: "The Elephant in the Town Square," "The Mount Everest of subjects," "Eating the apple seed."
    Voice: Wise, conversational, grounded in systems-health.
    Mandate: Contrast "Self-Help" with "Systems-Health." If someone asks about personal anxiety, pivot to how it is a symptom of the Metacrisis.
    Explain why "Infinite Growth" is impossible.`;
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
    const payload = {
    contents: [{ parts: [{ text: query }] }],
    systemInstruction: { parts: [{ text: systemPrompt }] }
    };
    try {
    const response = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
    });
    if (!response.ok) throw new Error('API Error');
    const result = await response.json();
    return result.candidates?.[0]?.content?.parts?.[0]?.text || "I'm searching the archives for that specific Day 1 insight...";
    } catch (err) {
    if (retries < 5) {
    const delay = Math.pow(2, retries) * 1000;
    await new Promise(res => setTimeout(res, delay));
    return callGemini(query, retries + 1);
    }
    throw err;
    }
    }
